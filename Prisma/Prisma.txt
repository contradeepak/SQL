// 1. Load Cart from Local Storage on startup
  const [cart, setCart] = useState(() => {
    const saved = localStorage.getItem("myCart");
    return saved ? JSON.parse(saved) : [];
  });

  const [view, setView] = useState("home"); // To switch between Product List and Cart

  // 2. Save to Local Storage whenever 'cart' updates
  useEffect(() => {
    localStorage.setItem("myCart", JSON.stringify(cart));
  }, [cart]);

  // 3. Add to Cart Logic
  const addToCart = (product) => {
    setCart([...cart, product]);
    alert(`${product.name} added!`);
  };

  // 4. Remove from Cart Logic (Optional but useful)
  const removeFromCart = (indexToRemove) => {
    setCart(cart.filter((_, index) => index !== indexToRemove));
